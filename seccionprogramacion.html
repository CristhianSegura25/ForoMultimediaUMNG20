function render(){

var t=0;

if(upArrow) {
t=0.1;
}
if(downArrow) {
t=-0.1;
}

//ROTACIÓN EN EJE X
if(xAxis){
var a = new THREE.Euler(t,0,0,'XYZ');
}
//ROTACIÓN EN EJE Y
else if(yAxis){

}
//ROTACIÓN EN EJE Z
else if (zAxis){

}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
public class SimulationEngine : MonoBehaviour
{
private GameObject mView = null;
private Vector3 mposicion_p;
private Vector3 mvelocidad_p;
private Vector3 A_1;
private Vector3 A_2;
private Vector3 A_3;
private Vector3 A_4;
private Vector3 mFuerza_1;
private Vector3 mFuerza_2;
private Vector3 mFuerza_3;
private Vector3 mFuerza_4;
private Vector3 mFuerza_total;

private float m;
private float k;
private float h;
private float mTime;
private float sTime;

private float carga_p;
private float carga_1;
private float carga_2;
private float carga_3;
private float carga_4;
private float [] f;
private float [] df;
// Start is called before the first frame update
void Start()
{
mTime=0;
sTime=0;
mView=GameObject.Find("ParticleView");

f=new float[6];
df=new float[6];
mposicion_p=new Vector3(-19.0f,-1.0f,0.0f);
mvelocidad_p=new Vector3(0.0f,0.0f,0.0f);
m=1.0f;
carga_p=1.0f;
A_1=new Vector3(-10.0f,0.0f,0.0f);
carga_1=150.0f;
A_2=new Vector3(-30.0f,0.0f,0.0f);
carga_2=-150.0f;
A_3=new Vector3(-20.0f,10.0f,0.0f);
carga_3=-150;
A_4=new Vector3(-20.0f,-10.0f,0.0f);
carga_4=150;

for(int i=0;i < 3;i++)
{
f[i]=mposicion_p[i];
f[i+3]=mvelocidad_p[i];
df[i]=0.0f;
df[i+3]=0.0f;
}
k=1.0f;
h=0.05f;


}
void Calculo_variables_instantaneas()
{
mFuerza_1=new Vector3(0.0f,0.0f,0.0f);
mFuerza_2=new Vector3(0.0f,0.0f,0.0f);
mFuerza_3=new Vector3(0.0f,0.0f,0.0f);
mFuerza_4=new Vector3(0.0f,0.0f,0.0f);
mFuerza_total=new Vector3(0.0f,0.0f,0.0f);

mposicion_p=new Vector3(f[0],f[1],f[2]);
carga_p=1.0f;
if(mposicion_p.x<=-20.0f)
{
carga_p=carga_p*-1;
}
if(carga_p==1)
{
mFuerza_1=mFuerza_1+(k*(carga_1*carga_p)/Mathf.Pow((Vector3.Magnitude((mposicion_p-A_1))),2)*(mposicion_p-A_1)/(Vector3.Magnitude((mposicion_p-A_1))));
mFuerza_2=mFuerza_2+(k*(carga_2*carga_p)/Mathf.Pow((Vector3.Magnitude((A_2-mposicion_p))),2)*(A_2-mposicion_p)/(Vector3.Magnitude((A_2-mposicion_p))));
mFuerza_3=mFuerza_3+(k*(carga_3*carga_p)/Mathf.Pow((Vector3.Magnitude((A_3-mposicion_p))),2)*(A_3-mposicion_p)/(Vector3.Magnitude((A_3-mposicion_p))));
mFuerza_4=mFuerza_4+(k*(carga_4*carga_p)/Mathf.Pow((Vector3.Magnitude((mposicion_p-A_4))),2)*(mposicion_p-A_4)/(Vector3.Magnitude((mposicion_p-A_4))));
Debug.Log(mposicion_p);
}
else
{
if(carga_p==-1)
{
mFuerza_1=mFuerza_1+(k*(carga_1*carga_p)/Mathf.Pow((Vector3.Magnitude((A_1-mposicion_p))),2)*(A_1-mposicion_p)/(Vector3.Magnitude((A_1-mposicion_p))));
mFuerza_2=mFuerza_2+(k*(carga_2*carga_p)/Mathf.Pow((Vector3.Magnitude((mposicion_p-A_2))),2)*(mposicion_p-A_2)/(Vector3.Magnitude((mposicion_p-A_2))));
mFuerza_3=mFuerza_3+(k*(carga_3*carga_p)/Mathf.Pow((Vector3.Magnitude((mposicion_p-A_3))),2)*(mposicion_p-A_3)/(Vector3.Magnitude((mposicion_p-A_3))));
mFuerza_4=mFuerza_4+(k*(carga_4*carga_p)/Mathf.Pow((Vector3.Magnitude((A_4-mposicion_p))),2)*(A_4-mposicion_p)/(Vector3.Magnitude((A_4-mposicion_p))));
Debug.Log(mposicion_p);
}
}
mFuerza_total=mFuerza_1+mFuerza_2+mFuerza_3+mFuerza_4;

for(int i=0;i < 3;i++)
{
df[i]=f[i+3];
df[i+3]=(mFuerza_total[i]/m);
}


}
void resolver_Ec()
{
for(int i=0;i<=5;i++)
{
f[i]=f[i]+df[i]*h;
}
sTime=sTime+h;
}

// Update is called once per frame
void Update()
{
mTime=Time.time;
Calculo_variables_instantaneas();
resolver_Ec();
Vector3 px=new Vector3(f[0],f[1],f[2]);
mView.transform.position=px;

}
}

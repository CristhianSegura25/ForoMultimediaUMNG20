using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using System;

public class Controller : MonoBehaviour
{
public GameObject[] Particle = new GameObject[2];
public GameObject[] Planos = new GameObject[6];
public Vector3[] Vel_Incial = new Vector3[4];
[SerializeField] float h;
private Vector3 Origin;
private Vector3 Gravedad;
private Vector3[] Pos = new Vector3[4];
private Vector3[] Vel = new Vector3[4];
private Vector3[] Fuerza = new Vector3[4];

/*Variables: Runge-Kutta*/
private float[,] Df = new float[4, 6];
private float[,] f = new float[4, 6];
private float[,,] Ki = new float[4, 4, 6];

/*Variables: Impulso*/
private Vector3[] Parallel = new Vector3[6];
private Vector3[] Impulso = new Vector3[4];
private Vector3[] Normal = new Vector3[6];
private Vector3[] DX = new Vector3[4];
private Vector3 Dist_P;
private float[] planoEC = new float[6];
private float J_Magnitude;
private float Particle_R;
private float Epsilom;
private float Miu;
private float m;

void Start()
{
for (int k = 0; k < 4; k++)
{
Pos[k] = Particle[k].transform.position;
Vel[k] = Vel_Incial[k];

f[k, 0] = Pos[k].x;
f[k, 1] = Pos[k].y;
f[k, 2] = Pos[k].z;
f[k, 3] = Vel[k].x;
f[k, 4] = Vel[k].y;
f[k, 5] = Vel[k].z;

for (int i = 0; i < 3; i++)
{
Df[k, i] = f[k, i + 3];
}
}

Origin = new Vector3(0.0f, 0.0f, 0.0f);
Gravedad = new Vector3(0.0f, 0.0f, 0.0f);
m = 1.0f;
Epsilom = 0.8f;
Miu = 0.2f;
}

void NyPCalculated(GameObject Plane, int i, int k)
{
Normal[i] = (Origin - Plane.transform.position).normalized;
Parallel[i] = Vel[k] - (Vector3.Dot(Vel[k], Normal[i])) * Normal[i];
Parallel[i] = Parallel[i].normalized;
planoEC[i] = Vector3.Dot(Normal[i], Pos[k]) +
(Plane.transform.position).magnitude;
}

void Update()
{
Particle_R = (Particle[1].transform.localScale.x) / 2;

for (int k = 0; k < 4; k++)

{
Fuerza[k] = Gravedad;
}

for (int k = 0; k < 4; k++)
{
Impulso[k] = Origin;
for (int i = 0; i < 6; i++)
{
NyPCalculated(Planos[i], i, k);
}

Impulso[k].x = m * f[k, 3];
Impulso[k].y = m * f[k, 4];
Impulso[k].z = m * f[k, 5];

for (int n = 0; n < 4; n++)
{
if (n != k)
{
Dist_P = Pos[k] - Pos[n];

if (Dist_P.magnitude <= 2 * Particle_R)
{
J_Magnitude = (1.0f + Epsilom) * m;

Impulso[k].x += -J_Magnitude *
((Vector3.Dot(Dist_P.normalized, Vel[k]) *
Dist_P.normalized.x) + (Miu * Parallel[n].x));

Impulso[k].y += -J_Magnitude *
((Vector3.Dot(Dist_P.normalized, Vel[k]) *
Dist_P.normalized.y) + (Miu * Parallel[n].y));

Impulso[k].z += -J_Magnitude *
((Vector3.Dot(Dist_P.normalized, Vel[k]) *
Dist_P.normalized.z) + (Miu * Parallel[n].z));

}
}
}

for (int n = 0; n < 6; n++)
{
if (planoEC[n] - Particle_R < 0) //Detección de la colisión
{
J_Magnitude = (1.0f + Epsilom) * m;

Impulso[k].x += -J_Magnitude * ((Vector3.Dot(Normal[n], Vel[k])
* Normal[n].x) + (Miu * Parallel[n].x));
Impulso[k].y += -J_Magnitude * ((Vector3.Dot(Normal[n], Vel[k])
* Normal[n].y) + (Miu * Parallel[n].y));
Impulso[k].z += -J_Magnitude * ((Vector3.Dot(Normal[n], Vel[k])
* Normal[n].z) + (Miu * Parallel[n].z));

DX[k] = (Particle_R - planoEC[n]) * Normal[n];

f[k, 0] += DX[k].x;
f[k, 1] += DX[k].y;
f[k, 2] += DX[k].z;

}
}

Df[k, 0] = (Impulso[k] / m).x;
Df[k, 1] = (Impulso[k] / m).y;
Df[k, 2] = (Impulso[k] / m).z;
Df[k, 3] = (Fuerza[k] / m).x;
Df[k, 4] = (Fuerza[k] / m).y;
Df[k, 5] = (Fuerza[k] / m).z;

/* Runge Kutta */

for (int i = 0; i < 6; i++)
{
Ki[k, 0, i] = Df[k, i];
Ki[k, 1, i] = (h / 2) * Df[k, i] + (Ki[k, 0, i] / 2);
Ki[k, 2, i] = (h / 2) * Df[k, i] + (Ki[k, 1, i] / 2);
Ki[k, 3, i] = Df[k, i] + (Ki[k, 2, i]);

f[k, i] += (h / 6) * (Ki[k, 0, i] + (2 * Ki[k, 1, i]) + (2 * Ki[k,
2, i]) + Ki[k, 3, i]);

}
for (int i = 3; i < 6; i++)
{

f[k, i] = Df[k, i - 3] + (h / 6) * (Ki[k, 0, i] + (2 * Ki[k, 1, i])
+ (2 * Ki[k, 2, i]) + Ki[k, 3, i]);

Df[k, i - 3] = f[k, i];
}

/* Actualización de datos */
Pos[k].x = f[k, 0];
Pos[k].y = f[k, 1];
Pos[k].z = f[k, 2];
Vel[k].x = f[k, 3];
Vel[k].y = f[k, 4];
Vel[k].z = f[k, 5];

Particle[k].transform.position = Pos[k];

Debug.Log("Velocidad " + k + ": " + f[k, 3] + ", " + f[k, 4] + ", " +
f[k, 5]);
Debug.Log("Posicion " + k + ": " + f[k, 0] + ", " + f[k, 1] + ", " +
f[k, 2]);

}
}
}

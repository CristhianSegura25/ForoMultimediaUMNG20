function init() {

camera = new THREE.PerspectiveCamera( 45, window.innerWidth / window.innerHeight, 1, 10000 );
camera.position.set( 500, 800, 1300 );
camera.lookAt( 0, 0, 0 );

scene = new THREE.Scene();
scene.background = new THREE.Color( 0xf0f0f0 );



cubeGeo = new THREE.BoxBufferGeometry( 50, 25, 50);
cilinderGeo = new THREE.CylinderBufferGeometry( 20, 20, 50, 75 );

//------------------------------------------------ Extrude

var shape = new THREE.Shape();
shape.moveTo(0,0);

shape.lineTo(25,50);
shape.lineTo(25,25);
shape.lineTo(50,25);
shape.lineTo(50,0);
shape.lineTo(0,0);

steps: 1,
amount: 50,
bevelEnabled: false,

extrude1 = new THREE.ExtrudeGeometry( shape, extrudeSettings );

var shape = new THREE.Shape();
shape.moveTo(0,0);
shape.lineTo(0,35);

objects.push( plane );

scene.add( ambientLight );

var directionalLight = new THREE.DirectionalLight( 0xffffff );

function onDocumentMouseDown( event ) { //---------------------- FunciÃ³n del mouse y voxeles

event.preventDefault();

mouse.set( ( event.clientX / window.innerWidth ) * 2 - 1, - ( event.clientY / window.innerHeight ) * 2 + 1 );

raycaster.setFromCamera( mouse, camera );

var intersects = raycaster.intersectObjects( objects );

if ( intersects.length > 0 ) {

var intersect = intersects[ 0 ];

// delete cube

if ( isShiftDown ) {

if ( intersect.object !== plane ) {

scene.remove( intersect.object );

objects.splice( objects.indexOf( intersect.object ), 1 );

}

// create cube

} else {

if (opcion == 1) {
voxel = new CSG(cubeGeo, cilinderGeo);
voxel = voxel.draw();
voxel.position.copy( intersect.point ).add( intersect.face.normal );
voxel.position.divideScalar( 27 ).floor().multiplyScalar( 25 ).addScalar( 25 );

objects.push( voxel );
